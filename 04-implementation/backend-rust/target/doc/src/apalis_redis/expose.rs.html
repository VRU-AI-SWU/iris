23    async fn stats(&self) -> Result<Stat, redis::RedisError> {
24        let mut conn = self.get_connection().clone();
25        let queue = self.get_config();
26        let script = r#"
27            local pending_jobs_set = KEYS[1]
28            local running_jobs_set = KEYS[2]
29            local dead_jobs_set = KEYS[3]
30            local failed_jobs_set = KEYS[4]
31            local success_jobs_set = KEYS[5]
32
33            local pending_count = redis.call('ZCARD', pending_jobs_set)
34            local running_count = redis.call('ZCARD', running_jobs_set)
35            local dead_count = redis.call('ZCARD', dead_jobs_set)
36            local failed_count = redis.call('ZCARD', failed_jobs_set)
37            local success_count = redis.call('ZCARD', success_jobs_set)
38
39            return {pending_count, running_count, dead_count, failed_count, success_count}
40    "#;
41
42        let keys = vec![
43            queue.inflight_jobs_set().to_string(),
44            queue.active_jobs_list().to_string(),
45            queue.dead_jobs_set().to_string(),
46            queue.failed_jobs_set().to_string(),
47            queue.done_jobs_set().to_string(),
48        ];
49
50        let results: Vec<usize> = redis::cmd("EVAL")
51            .arg(script)
52            .arg(keys.len().to_string())
53            .arg(keys)
54            .query_async(&mut conn)
55            .await?;
56
57        Ok(Stat {
58            pending: results[0],
59            running: results[1],
60            dead: results[2],
61            failed: results[3],
62            success: results[4],
63        })
64    }
65    async fn list_jobs(
66        &self,
67        status: &State,
68        page: i32,
69    ) -> Result<Vec<Self::Request>, redis::RedisError> {
70        let mut conn = self.get_connection().clone();
71        let queue = self.get_config();
72        match status {
73            State::Pending | State::Scheduled => {
74                let active_jobs_list = &queue.active_jobs_list();
75                let job_data_hash = &queue.job_data_hash();
76                let ids: Vec<String> = redis::cmd("LRANGE")
77                    .arg(active_jobs_list)
78                    .arg(((page - 1) * 10).to_string())
79                    .arg((page * 10).to_string())
80                    .query_async(&mut conn)
81                    .await?;
82
83                if ids.is_empty() {
84                    return Ok(Vec::new());
85                }
86                let data: Option<Value> = redis::cmd("HMGET")
87                    .arg(job_data_hash)
88                    .arg(&ids)
89                    .query_async(&mut conn)
90                    .await?;
91
92                let jobs: Vec<Request<T, RedisContext>> =
93                    deserialize_multiple_jobs::<_, RedisCodec>(data.as_ref()).unwrap();
94                Ok(jobs)
95            }
96            State::Running => {
97                let consumers_set = &queue.consumers_set();
98                let job_data_hash = &queue.job_data_hash();
99                let workers: Vec<String> = redis::cmd("ZRANGE")
100                    .arg(consumers_set)
101                    .arg("0")
102                    .arg("-1")
103                    .query_async(&mut conn)
104                    .await?;
105
106                if workers.is_empty() {
107                    return Ok(Vec::new());
108                }
109                let mut all_jobs = Vec::new();
110                for worker in workers {
111                    let ids: Vec<String> = redis::cmd("SMEMBERS")
112                        .arg(&worker)
113                        .query_async(&mut conn)
114                        .await?;
115
116                    if ids.is_empty() {
117                        continue;
118                    };
119                    let data: Option<Value> = redis::cmd("HMGET")
120                        .arg(job_data_hash.clone())
121                        .arg(&ids)
122                        .query_async(&mut conn)
123                        .await?;
124
125                    let jobs: Vec<Request<T, RedisContext>> =
126                        deserialize_multiple_jobs::<_, RedisCodec>(data.as_ref()).unwrap();
127                    all_jobs.extend(jobs);
128                }
129
130                Ok(all_jobs)
131            }
132            State::Done => {
133                let done_jobs_set = &queue.done_jobs_set();
134                let job_data_hash = &queue.job_data_hash();
135                let ids: Vec<String> = redis::cmd("ZRANGE")
136                    .arg(done_jobs_set)
137                    .arg(((page - 1) * 10).to_string())
138                    .arg((page * 10).to_string())
139                    .query_async(&mut conn)
140                    .await?;
141
142                if ids.is_empty() {
143                    return Ok(Vec::new());
144                }
145                let data: Option<Value> = redis::cmd("HMGET")
146                    .arg(job_data_hash)
147                    .arg(&ids)
148                    .query_async(&mut conn)
149                    .await?;
150
151                let jobs: Vec<Request<T, RedisContext>> =
152                    deserialize_multiple_jobs::<_, RedisCodec>(data.as_ref()).unwrap();
153                Ok(jobs)
154            }
155            // State::Retry => Ok(Vec::new()),
156            State::Failed => {
157                let failed_jobs_set = &queue.failed_jobs_set();
158                let job_data_hash = &queue.job_data_hash();
159                let ids: Vec<String> = redis::cmd("ZRANGE")
160                    .arg(failed_jobs_set)
161                    .arg(((page - 1) * 10).to_string())
162                    .arg((page * 10).to_string())
163                    .query_async(&mut conn)
164                    .await?;
165                if ids.is_empty() {
166                    return Ok(Vec::new());
167                }
168                let data: Option<Value> = redis::cmd("HMGET")
169                    .arg(job_data_hash)
170                    .arg(&ids)
171                    .query_async(&mut conn)
172                    .await?;
173                let jobs: Vec<Request<T, RedisContext>> =
174                    deserialize_multiple_jobs::<_, RedisCodec>(data.as_ref()).unwrap();
175
176                Ok(jobs)
177            }
178            State::Killed => {
179                let dead_jobs_set = &queue.dead_jobs_set();
180                let job_data_hash = &queue.job_data_hash();
181                let ids: Vec<String> = redis::cmd("ZRANGE")
182                    .arg(dead_jobs_set)
183                    .arg(((page - 1) * 10).to_string())
184                    .arg((page * 10).to_string())
185                    .query_async(&mut conn)
186                    .await?;
187
188                if ids.is_empty() {
189                    return Ok(Vec::new());
190                }
191                let data: Option<Value> = redis::cmd("HMGET")
192                    .arg(job_data_hash)
193                    .arg(&ids)
194                    .query_async(&mut conn)
195                    .await?;
196
197                let jobs: Vec<Request<T, RedisContext>> =
198                    deserialize_multiple_jobs::<_, RedisCodec>(data.as_ref()).unwrap();
199
200                Ok(jobs)
201            }
202        }
203    }
204    async fn list_workers(&self) -> Result<Vec<Worker<WorkerState>>, redis::RedisError> {
205        let queue = self.get_config();
206        let consumers_set = &queue.consumers_set();
207        let mut conn = self.get_connection().clone();
208        let workers: Vec<String> = redis::cmd("ZRANGE")
209            .arg(consumers_set)
210            .arg("0")
211            .arg("-1")
212            .query_async(&mut conn)
213            .await?;
214        Ok(workers
215            .into_iter()
216            .map(|w| {
217                Worker::new(
218                    WorkerId::new(w.replace(&format!("{}:", &queue.inflight_jobs_set()), "")),
219                    WorkerState::new::<Self>(queue.get_namespace().to_owned()),
220                )
221            })
222            .collect())
223    }

226fn deserialize_multiple_jobs<T, C: Codec<Compact = Vec<u8>>>(
227    jobs: Option<&Value>,
228) -> Option<Vec<Request<T, RedisContext>>>
229where
230    T: DeserializeOwned,
231{
232    let jobs = match jobs {
233        None => None,
234        Some(Value::Array(val)) => Some(val),
235        _ => {
236            // error!(
237            //     "Decoding Message Failed: {:?}",
238            //     "unknown result type for next message"
239            // );
240            None
241        }
242    };
243
244    jobs.map(|values| {
245        values
246            .iter()
247            .filter_map(|v| match v {
248                Value::BulkString(data) => {
249                    let inner = C::decode(data.to_vec())
250                        .map_err(|e| (ErrorKind::IoError, "Decode error", e.into().to_string()))
251                        .unwrap();
252                    Some(inner)
253                }
254                _ => None,
255            })
256            .collect()
257    })
